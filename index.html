const { Client, GatewayIntentBits, EmbedBuilder } = require('discord.js'); const admin = require('firebase-admin'); // חיבור ל-Firebase const serviceAccount = require("./serviceAccountKey.json"); admin.initializeApp({ credential: admin.credential.cert(serviceAccount), databaseURL: "https://tgi-webs-default-rtdb.europe-west1.firebasedatabase.app" }); const db = admin.database(); const client = new Client({ intents: [GatewayIntentBits.Guilds, GatewayIntentBits.GuildMessages, GatewayIntentBits.MessageContent] }); client.on('ready', () => console.log(`🚀 הבוט ${client.user.tag} מוכן לעבודה!`)); client.on('messageCreate', async (message) => { if (message.author.bot) return; // פקודה: שלח [ID] [תשובה] if (message.content.startsWith('שלח')) { const args = message.content.split(' '); const questionId = args[1]; const answerText = args.slice(2).join(' '); if (!questionId || !answerText) { return message.reply("❌ פורמט לא תקין. רשום: `שלח [מזהה] [תשובה]`"); } try { const questionRef = db.ref(`public_questions/${questionId}`); const snapshot = await questionRef.once('value'); if (!snapshot.exists()) { return message.reply("❓ לא מצאתי שאלה עם המזהה הזה."); } const questionData = snapshot.val(); await questionRef.update({ answer: answerText }); // הודעת אישור יפה בדיסקורד const embed = new EmbedBuilder() .setTitle('✅ התשובה פורסמה באתר!') .setDescription(`שלחת תשובה ל-**${questionData.name}**`) .addFields( { name: 'השאלה המקורית:', value: questionData.text }, { name: 'התשובה שלך:', value: answerText } ) .setColor('#00ff00'); message.reply({ embeds: [embed] }); } catch (error) { console.error(error); message.reply("❌ תקלה בעדכון ה-Firebase."); } } }); client.login('כאן_שים_את_הטוקן_שלך');
